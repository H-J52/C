```c
#include <stdio.h>
//할인율 구하기

int solution(int price, char* grade) {
	int answer = 0;

	//grade가 "S"일 시 할인율 5%
	if (grade == "S") {
		answer = price * 0.95;
	}
	//grade가 "G"일 시 할인율 10%
	if (grade == "G") {
		answer = price * 0.9;
	}
	//grade가 "V"일 시 할인율 15%
	if (grade == "V") {
		answer = price * 0.85;
	}

	return answer;
}


int main(void) {
	int price1 = 500;
	char* grade1 = "V";
	int ret1 = solution(price1, grade1);

	printf("solution 함수의 반환 값은 %d 입니다.\n", ret1);

	int price2 = 96900;
	char* grade2 = "S";
	int ret2 = solution(price2, grade2);

	printf("solution 함수의 반환 값은 %d 입니다.\n", ret2);
}
```
