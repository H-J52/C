#include <stdio.h>
//소수점 구하는 코드

int get_integer(void);
int is_prime(int n);

int main(void) {
    int n;
    n = get_integer();
    is_prime(n);
}

int get_integer(void) {
    int n;
    printf("정수를 입력해주세요: ");
    scanf_s("%d", &n);
    return n;
}

int is_prime(int n) {
    int counter = 0;

    for (int i = 2; i < n; i++) {
        for (int j = 1; j <= i; j++) {
            if (i % j == 0) {
                counter += 1;
            }
        }
        if (counter == 2) {
            printf("%d ", i);
        }
        counter = 0;
    }
    return 0;
}
