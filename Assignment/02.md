```c
#include <stdlib.h> //표준 라이브러리의 일부 함수를 사용하기 위한 헤더 파일
#include <stdio.h> //표준 입출력 함수를 사용하기 위한 헤더 파일
#include <conio.h> //특정 키 입력을 처리하기 위한 헤더 파일 -> (Windows 환경에서만 사용 가능)
#include <time.h> //시간 함수를 사용하기 위한 헤더 파일

void disp_car(int car_number, int distance); //자동차의 정보를 표시(자동차 번호, 거리)하는 함수의 선언

int main(void) {
    int i;
    int car1_dist = 0, car2_dist = 0; //각 자동차의 이동 거리를 저장하는 변수

    srand((unsigned)time(NULL)); //시간을 기반으로 하는 시드 값을 사용하며 난수 생성을 초기화하는 코드

    for (i = 0; i < 6; i++) { //총 6번 반복
        car1_dist += rand() % 100; //0부터 99까지의 난수를 발생시켜 car1_dist에 더함
        car2_dist += rand() % 100; //0부터 99까지의 난수를 발생시켜 car2_dist에 더함

        disp_car(1, car1_dist); //자동차 1의 번호와 이동 거리를 disp_car 함수로 전달
        disp_car(2, car2_dist); //자동차 2의 번호와 이동 거리를 disp_car 함수로 전달

        printf("---------------------\n"); //구분선 출력

        _getch(); //특정 키 입력 대기 -> (Windows 환경에서만 사용 가능)
    }

    return 0;
}

void disp_car(int car_number, int distance) {
    int i;

    printf("CAR #%d:", car_number); //자동차 번호 출력

    for (i = 0; i < distance / 10; i++) { //i 변수는 0부터 시작하여 distance를 10으로 나눈 몫만큼 반복
        printf("*"); //이동 거리를 10으로 나눈 만큼 "*" 출력
    }

    printf("\n"); //줄 바꿈
}

/* 작동 순서

1. disp_car 함수와 main 함수를 정의한다.

2. main 함수에서 사용할 변수 i, car1_dist, car2_dist를 선언하고, srand 함수를 사용하여 난수 생성을 초기화한다. (이때 시간을 기반으로 한 시드 값을 사용한다.)

3. for 루프를 사용하여 경주 시뮬레이션을 6번 반복한다. (i 변수는 반복 횟수를 나타낸다.)

4. 각 반복에서, car1_dist와 car2_dist 변수에 랜덤한 값을 더한다. rand() % 100은 0부터 99까지의 난수를 발생시키도록 제한하는데, 이를 통해 각 자동차의 이동 거리를 결정한다.

5. disp_car 함수를 호출하여 자동차 번호와 이동 거리를 전달하고, 각 자동차의 이동 거리에 따라 별표(*)를 출력한다. (이때 disp_car 함수에서 distance를 10으로 나눈 값만큼 별표를 출력하게 된다.)

6. 경주 시뮬레이션 단계마다 구분선(---------------------)을 출력한다.

7. getch 함수를 호출하여 사용자가 키를 누를 때까지 대기한다.

8. for 루프가 반복되면 다음 경주 시뮬레이션 단계로 넘어가고, 총 6번의 경주가 완료되면 main 함수에서 return 0;을 통해 프로그램을 종료한다.*/
```
