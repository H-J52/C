# 73 슬라이드 정리
```c
#include <math.h> //삼각함수를 사용하기 위한 헤더 파일
#include <stdio.h> //표준 입출력 함수를 사용하기 위한 헤더 파일

int main(void) {
    double pi = 3.1415926535;
    double x, y;

    x = pi / 2; //x에 pi/2 값을 저장
    y = sin(x); //x의 사인 값 계산 후 y에 저장
    printf("sin(%f) = %f\n", x, y); //sin(1.570796) = 1.000000

    y = cos(x); //x의 코사인 값 계산 후 y에 저장
    printf("cos(%f) = %f\n", x, y); //cos(1.570796) = 0.000000

    return 0;
}
```
